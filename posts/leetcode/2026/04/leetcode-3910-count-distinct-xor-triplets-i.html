Today we solve LeetCode 3910 - Count Distinct XOR Triplets I.

Source: https://leetcode.com/problems/count-distinct-xor-triplets-i/

English

We enumerate all ordered index triplets with i ≤ j ≤ k, compute nums[i] ^ nums[j] ^ nums[k], and mark the value as seen. The answer is the number of distinct seen XOR values.

Because values are bounded in this problem, a boolean array is enough and very fast in practice.

class Solution {
    public int countTriplets(int[] nums) {
        boolean[] seen = new boolean[2048];
        int n = nums.length;
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                for (int k = j; k < n; k++) {
                    seen[nums[i] ^ nums[j] ^ nums[k]] = true;
                }
            }
        }
        int ans = 0;
        for (boolean v : seen) if (v) ans++;
        return ans;
    }
}
func countTriplets(nums []int) int {
    seen := make([]bool, 2048)
    n := len(nums)
    for i := 0; i < n; i++ {
        for j := i; j < n; j++ {
            for k := j; k < n; k++ {
                seen[nums[i]^nums[j]^nums[k]] = true
            }
        }
    }
    ans := 0
    for _, v := range seen {
        if v { ans++ }
    }
    return ans
}
class Solution {
public:
    int countTriplets(vector<int>& nums) {
        vector<bool> seen(2048, false);
        int n = nums.size();
        for (int i = 0; i < n; i++) {
            for (int j = i; j < n; j++) {
                for (int k = j; k < n; k++) {
                    seen[nums[i] ^ nums[j] ^ nums[k]] = true;
                }
            }
        }
        return count(seen.begin(), seen.end(), true);
    }
};
class Solution:
    def countTriplets(self, nums: List[int]) -> int:
        seen = [False] * 2048
        n = len(nums)
        for i in range(n):
            for j in range(i, n):
                for k in range(j, n):
                    seen[nums[i] ^ nums[j] ^ nums[k]] = True
        return sum(seen)
/**
 * @param {number[]} nums
 * @return {number}
 */
var countTriplets = function(nums) {
  const seen = new Array(2048).fill(false);
  const n = nums.length;
  for (let i = 0; i < n; i++) {
    for (let j = i; j < n; j++) {
      for (let k = j; k < n; k++) {
        seen[nums[i] ^ nums[j] ^ nums[k]] = true;
      }
    }
  }
  return seen.filter(Boolean).length;
};

中文

我们直接枚举所有满足 i ≤ j ≤ k 的三元组，计算 nums[i] ^ nums[j] ^ nums[k]，并把结果标记到 visited 数组中。最后统计出现过多少个不同的异或值。

由于题目数值范围较小，使用布尔数组比哈希集合更直接，常数也更小。
