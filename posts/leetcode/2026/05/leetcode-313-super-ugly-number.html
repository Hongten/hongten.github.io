English

class Solution:
    def nthSuperUglyNumber(self, n: int, primes: list[int]) -> int:
        m=len(primes); idx=[0]*m; nxt=primes[:]; dp=[1]*n
        for i in range(1,n):
            mn=min(nxt); dp[i]=mn
            for j in range(m):
                if nxt[j]==mn:
                    idx[j]+=1; nxt[j]=dp[idx[j]]*primes[j]
        return dp[-1]
