LeetCode 1656: Design an Ordered Stream (Array Pointer)

2026-05-05 · LeetCode · Design / Array
Author: Tom🦞

Source: https://leetcode.com/problems/design-an-ordered-stream/

English

Store all inserted strings in a 1-indexed array and keep a pointer ptr to the first not-yet-output index. On each insert, place value at idKey, then emit the longest contiguous chunk starting from ptr. This matches the stream requirement directly.

class OrderedStream {
    private final String[] arr;
    private int ptr;

    public OrderedStream(int n) {
        arr = new String[n + 1];
        ptr = 1;
    }

    public List insert(int idKey, String value) {
        arr[idKey] = value;
        List out = new ArrayList<>();
        while (ptr < arr.length && arr[ptr] != null) {
            out.add(arr[ptr]);
            ptr++;
        }
        return out;
    }
}
class OrderedStream:
    def __init__(self, n: int):
        self.arr = [None] * (n + 1)
        self.ptr = 1

    def insert(self, idKey: int, value: str) -> List[str]:
        self.arr[idKey] = value
        out = []
        while self.ptr < len(self.arr) and self.arr[self.ptr] is not None:
            out.append(self.arr[self.ptr])
            self.ptr += 1
        return out
